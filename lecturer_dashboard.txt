        let allStudents = [], allAttendance = [];

        async function loadData() {
            const studentsSnap = await db.collection('students').get();
            allStudents = [];
            studentsSnap.forEach(doc => allStudents.push({ id: doc.id, ...doc.data() }));
            const attendanceSnap = await db.collection('attendance').get();
            allAttendance = [];
            attendanceSnap.forEach(doc => allAttendance.push({ id: doc.id, ...doc.data() }));
            renderClasses();
        }

        function renderClasses() {
            const classFilter = document.getElementById('classFilter').value;
            const search = document.getElementById('searchInput').value.toLowerCase();
            const byClass = {};
            allStudents.forEach(s => { let cls = s.class || "No Class"; if (!byClass[cls]) byClass[cls] = []; byClass[cls].push(s); });
            const attendanceMap = {};
            allAttendance.forEach(a => attendanceMap[a.student_id] = a);

            let classes = Object.keys(byClass).sort();
            if (classFilter !== 'all') classes = classes.filter(c => c === classFilter);
            let html = '';
            for (let cls of classes) {
                let students = byClass[cls];
                if (search) students = students.filter(s => s.student_name.toLowerCase().includes(search));
                if (!students.length) continue;
                let presentCount = 0, lateCount = 0;
                let studentsHtml = '';
                students.forEach(s => {
                    const att = attendanceMap[s.student_id];
                    let status = 'absent';
                    if (att) {
                        status = att.status; // 'present' or 'late'
                    }
                    if (status === 'present') presentCount++;
                    else if (status === 'late') lateCount++;
                    let statusClass = status === 'present' ? 'present' : (status === 'late' ? 'late' : 'absent');
                    let statusText = status === 'present' ? 'Present' : (status === 'late' ? 'Late' : 'Absent');
                    studentsHtml += `
                        <div class="student-row">
                            <span>${s.student_name}</span>
                            <span class="status-badge status-${statusClass}">
                                <i class="fas ${status === 'present' ? 'fa-check-circle' : (status === 'late' ? 'fa-clock' : 'fa-times-circle')}"></i> ${statusText}
                            </span>
                        </div>
                    `;
                });
                html += `
                    <div class="card">
                        <div class="card-header"><h3>📚 ${cls}</h3><p>${lecturerNames[cls] || "Staff"}</p></div>
                        <div class="students-list">${studentsHtml}</div>
                        <div class="card-footer"><strong>${presentCount} present</strong> ${lateCount ? `| ${lateCount} late` : ''} / ${students.length} total</div>
                    </div>
                `;
            }
            document.getElementById('lecturerGrid').innerHTML = html || '<div class="no-data">No data</div>';
        }

        document.getElementById('classFilter').addEventListener('change', renderClasses);
        document.getElementById('searchInput').addEventListener('input', renderClasses);
        loadData();
        setInterval(loadData, 30000);
